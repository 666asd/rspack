```mjs title=main.mjs
// ./index.js
it('should support dyn import with no conflicting exports', async () => {
	const { foo, fooFn } = await import("./shared.mjs")
	const { bar, barFn } = await import("./shared.mjs")

	expect(foo).toBe(1)
	expect(bar).toBe(2)

	expect(fooFn()).toBe(1)
	expect(barFn()).toBe(2)
})

export {};

```
