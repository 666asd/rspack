// ./index.js
it("should merge duplicate externals inside a dynamic multi-module chunk", async () => {
	const modA = await import("./shared.mjs").then(m => m.a_namespaceObject);
	const modB = await import("./shared.mjs").then(m => m.b_namespaceObject);

	expect(modA.fsDefault).toBeDefined();
	expect(modA.readFileSync).toBeDefined();
	expect(modB.readFile).toBe(modA.readFile);
	expect(modB.marker).toBe("b");
});

export {};

(() => {
__rspack_require.d = (exports, getters, values) => {
	var define = (defs, kind) => {
		for(var key in defs) {
			if(__rspack_require.o(defs, key) && !__rspack_require.o(exports, key)) {
				Object.defineProperty(exports, key, { enumerable: true, [kind]: defs[key] });
			}
		}
	};
	define(getters, "get");
	define(values, "value");
};
})();
