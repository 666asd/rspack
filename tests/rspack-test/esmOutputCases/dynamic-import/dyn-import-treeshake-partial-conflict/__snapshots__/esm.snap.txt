```mjs title=main.mjs
// ./index.js
it('should skip namespace when tree-shaking removes one side of a conflict', async () => {
	// Both modules export `value`, but only a.value is used.
	// b.value is tree-shaken, so the conflict disappears.
	const { value, foo } = await import("./shared.mjs")
	const { bar } = await import("./shared.mjs")

	expect(value).toBe(1)
	expect(foo).toBe('a-foo')
	expect(bar).toBe('b-bar')
})

export {};

```

```mjs title=shared.mjs
// ./a.js
const a_value = 1;
const a_foo = 'a-foo';

// ./b.js
const b_value = 2;
const b_bar = 'b-bar';

export { a_foo as foo, a_value as value, b_bar as bar };
```
