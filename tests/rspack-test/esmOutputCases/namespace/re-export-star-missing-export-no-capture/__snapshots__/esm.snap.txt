// The require scope
var __rspack_require = {};

// webpack/runtime/make_namespace_object
(() => {
// define __esModule on exports
__rspack_require.r = (__rspack_exports) => {
	if(typeof Symbol !== 'undefined' && Symbol.toStringTag) {
		Object.defineProperty(__rspack_exports, Symbol.toStringTag, { value: 'Module' });
	}
	Object.defineProperty(__rspack_exports, '__esModule', { value: true });
};

})();

export { __rspack_require };
