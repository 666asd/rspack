(() => {
__rspack_require.d = (exports, getters, values) => {
	var define = (defs, kind) => {
		for(var key in defs) {
			if(__rspack_require.o(defs, key) && !__rspack_require.o(exports, key)) {
				Object.defineProperty(exports, key, { enumerable: true, [kind]: defs[key] });
			}
		}
	};
	define(getters, "get");
	define(values, "value");
};
})();

	if(typeof Symbol !== 'undefined' && Symbol.toStringTag) {
